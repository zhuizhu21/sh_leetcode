class Solution {
public:
    bool isValid(string s) {
        int len=s.length();
        if(len%2!=0)
            return false;
        if(len==0)
            return true;
        map<char,char>m;
        vector<char>v;
        m['}']='{';
        m[']']='[';
        m[')']='(';
        for(int i=0;i<len;i++){
            if(s[i]=='{'||s[i]=='['||s[i]=='('){
                v.push_back(s[i]);
            }else if (s[i]==' ') {
                continue;
            }else {
                if(v.size()>0&&v[v.size()-1]==m[s[i]]){
                    v.pop_back();
                }else {
                    v.push_back(s[i]);
                }
            }
        }
        if(v.size()==0)
            return true;
        else
            return false;
    }
};
